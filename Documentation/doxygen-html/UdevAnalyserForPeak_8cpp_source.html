143 void UdevAnalyserForPeak::m_createUdevPortMap( void ){
144  LOG(Log::TRC, m_logItHandleSock) << "peak doing udev calls to discover PEAK socketcan device mapping";
145 
146  std::vector<std::string> links1;
147  std::vector<std::string> links2;
148 
149  // just the real devices, not the symlinks
150  std::string cmd0 = "ls -l /dev/pcanusb* | grep -v \" -> \" | awk '{print $10}' ";
151  execcommand_ns::ExecCommand exec0( cmd0 );
152  const execcommand_ns::ExecCommand::CmdResults results0 = exec0.getResults();
153  //LOG(Log::TRC, m_logItHandleSock) << "peak " << exec0;
154 
155  // get the symlinks for each device and the first port
156  for ( unsigned int i = 0; i < results0.size(); i++ ){
157  std::string cmd1 = std::string("/sbin/udevadm info -q symlink ") + results0[ i ] + std::string(" | grep \"devid=\"");
158  execcommand_ns::ExecCommand exec1( cmd1 );
159  execcommand_ns::ExecCommand::CmdResults results1 = exec1.getResults();
160  //LOG(Log::TRC, m_logItHandleSock) << "peak " << exec1;
161  for ( unsigned k = 0; k < results1.size(); k++ ){
162  links1.push_back( results1[ k ] );
163  }
164  }
165  // get the links of the other ports
166  for ( unsigned int i = 0; i < results0.size(); i++ ){
167  std::string cmd2 = std::string("/sbin/udevadm info -q symlink ") + results0[ i ] + std::string(" | grep -v \"devid=\"");
168  execcommand_ns::ExecCommand exec2( cmd2 );
169  execcommand_ns::ExecCommand::CmdResults results2 = exec2.getResults();
170  //LOG(Log::TRC, m_logItHandleSock) << "peak " << exec2;
171  for ( unsigned k = 0; k < results2.size(); k++ ){
172  links2.push_back( results2[ k ] );
173  }
174  }
175 
185  for ( unsigned int i = 0; i < links1.size(); i++ ){
186  PEAK_device_t peak;
187  peak.deviceID = m_peakDeviceId( links1[ i ] );
188  peak.systemDeviceIndex = m_peakSystemDeviceIndex( links1[ i ] );
189  peak.localCanPort = m_peakLocalCanPort( links1[ i ] );
190  peak.peakDriverNumber = m_peakDriverNumber( links1[ i ] );
191  peak.link = links1[ i ];
192  m_peak_v.push_back( peak );
193  }
194  for ( unsigned int i = 0; i < links2.size(); i++ ){
195  PEAK_device_t peak;
196  peak.systemDeviceIndex = m_peakSystemDeviceIndex( links2[ i ] );
197  peak.deviceID = m_peakDeviceIdFromSystemDeviceIndex( peak.systemDeviceIndex );
198  peak.localCanPort = m_peakLocalCanPort( links2[ i ] );
199  peak.peakDriverNumber = m_peakDriverNumber( links2[ i ] );
200  peak.link = links2[ i ];
201  m_peak_v.push_back( peak );
202  }
203 
208  sort( m_peak_v.begin(), m_peak_v.end(), UdevAnalyserForPeak::m_peakStructCompare );
209  for ( unsigned i = 0; i < m_peak_v.size(); i++ ){
210  m_peak_v[ i ].socketNumber = i;
211  }
212  // showMap();
213 }
