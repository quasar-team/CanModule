109 DWORD WINAPI PKCanScan::CanScanControlThread(LPVOID pCanScan)
110 {
111  PKCanScan *pkCanScanPointer = reinterpret_cast<PKCanScan *>(pCanScan);
112  TPCANHandle tpcanHandler = pkCanScanPointer->m_pkCanHandle;
113  MLOGPK(DBG,pkCanScanPointer) << "CanScanControlThread Started. m_CanScanThreadShutdownFlag = [" << pkCanScanPointer->m_CanScanThreadRunEnableFlag <<"]";
114  pkCanScanPointer->m_CanScanThreadRunEnableFlag = true;
115  while ( pkCanScanPointer->m_CanScanThreadRunEnableFlag ) {
116  TPCANMsg tpcanMessage;
117  TPCANTimestamp tpcanTimestamp;
118  TPCANStatus tpcanStatus = CAN_Read(tpcanHandler,&tpcanMessage,&tpcanTimestamp);
119  if ( tpcanStatus == PCAN_ERROR_OK ) {
120  CanMsgStruct canMessage;
121  for(int i=0; i < 8; i++) {
122  canMessage.c_data[i] = tpcanMessage.DATA[i];
123  }
124  canMessage.c_dlc = tpcanMessage.LEN;
125  canMessage.c_id = tpcanMessage.ID;
126  canMessage.c_ff = tpcanMessage.MSGTYPE;
127 
128  unsigned long long mmsec = 0xFFFFFFFFUL;
129  mmsec = mmsec * 1000;
130  mmsec = mmsec * tpcanTimestamp.millis_overflow;
131  mmsec = mmsec + tpcanTimestamp.micros;
132  mmsec = mmsec + 1000 * tpcanTimestamp.millis;
133 
134  canMessage.c_time.tv_usec = mmsec % 1000000UL;
135  canMessage.c_time.tv_sec = long(mmsec / 1000000UL);
136  pkCanScanPointer->canMessageCame(canMessage);
137  pkCanScanPointer->m_statistics.onReceive( canMessage.c_dlc );
138  pkCanScanPointer->m_statistics.setTimeSinceReceived();

140  // we can reset the reconnectionTimeout here, since we have received a message
141  pkCanScanPointer->resetTimeoutOnReception();
142  } else {
143  if (tpcanStatus & PCAN_ERROR_QRCVEMPTY) {
144  // timeout
150  CanModule::ReconnectAutoCondition rcond = pkCanScanPointer->getReconnectCondition();
151  CanModule::ReconnectAction ract = pkCanScanPointer->getReconnectAction();

412 bool PKCanScan::sendMessage(short cobID, unsigned char len, unsigned char *message, bool rtr)
413 {
414  MLOGPK(DBG,this) << "Sending message: [" << ( message == 0 ? "" : (const char *) message) << "], cobID: [" << cobID << "], Message Length: [" << static_cast<int>(len) << "]";
415 
416  TPCANStatus tpcanStatus;
417  TPCANMsg tpcanMessage;
418  tpcanMessage.ID = cobID;
419  tpcanMessage.MSGTYPE = PCAN_MESSAGE_STANDARD;
420  if(rtr) {
421  tpcanMessage.MSGTYPE = PCAN_MESSAGE_RTR;
422  }
423  int lengthOfUnsentData, lengthToBeSent;
424  lengthOfUnsentData = len;
425  do {
426  //In every iteration of this loop a piece of message is sent of maximun 8 chars
427  //To keep track of how much message is left we use vars lengthOfUnsentData and lengthToBeSent
428  if (lengthOfUnsentData > 8) {
429  lengthToBeSent = 8;
430  lengthOfUnsentData = lengthOfUnsentData - 8;
431  } else {
432  lengthToBeSent = lengthOfUnsentData;
433  }
434  tpcanMessage.LEN = lengthToBeSent;
435 
436  memcpy(tpcanMessage.DATA,message,lengthToBeSent);
437  tpcanStatus = CAN_Write(m_pkCanHandle, &tpcanMessage);
438  if (tpcanStatus != PCAN_ERROR_OK) {
439 
440  // here we should see if we need to reconnect
441  switch( m_reconnectCondition ){
442  case CanModule::ReconnectAutoCondition::sendFail: {
443  MLOGPK(WRN, this) << " detected a sendFail, triggerCounter= " << m_triggerCounter
444  << " failedSendCounter= " << m_failedSendCounter;
445  m_triggerCounter--;
446  break;
447  }
448  case CanModule::ReconnectAutoCondition::never:
449  default:{
450  m_triggerCounter = m_failedSendCounter;
451  break;
452  }
453  }
454 
455  switch ( m_reconnectAction ){
456  case CanModule::ReconnectAction::allBusesOnBridge: {
457  if ( m_triggerCounter <= 0 ){
458  MLOGPK(INF, this) << " reconnect condition " << (int) m_reconnectCondition
459  << reconnectConditionString(m_reconnectCondition)
460  << " triggered action " << (int) m_reconnectAction
461  << reconnectActionString(m_reconnectAction)
462  << " not yet implemented";
463  m_triggerCounter = m_failedSendCounter;
464  }
465  break;
466  }
467  case CanModule::ReconnectAction::singleBus: {
468  MLOGPK(INF, this) << " reconnect condition " << (int) m_reconnectCondition
469  << reconnectConditionString(m_reconnectCondition)
470  << " triggered action " << (int) m_reconnectAction
471  << reconnectActionString(m_reconnectAction);
472  if ( m_triggerCounter <= 0 ){
473  stopBus();
474  createBus(m_busName, m_busParameters );
475  MLOGPK(TRC, this) << "reconnect one CAN port m_pkCanHandle= " << m_pkCanHandle;
476  m_triggerCounter = m_failedSendCounter;
477  }
478  break;
479  }
480  default: {
481  break;
482  }
483  }
484 
485 
486  return sendErrorCode(tpcanStatus);
487  }
488  m_statistics.onTransmit( tpcanMessage.LEN );
489  m_statistics.setTimeSinceTransmitted();
490 
491  message = message + lengthToBeSent;
492  }
493  while (lengthOfUnsentData > 8);
494  return sendErrorCode(tpcanStatus);
495 }
