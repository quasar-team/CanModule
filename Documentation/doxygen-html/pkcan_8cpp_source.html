461

463  triggerReconnectionThread();

576  CanModule::ReconnectAction raction = pkCanScanPointer->getReconnectAction();

581  ss << std::this_thread::get_id();
582  _tid = ss.str();
583  }

585 
586  // need some sync to the main thread to be sure it is up and the sock is created: wait first time for init
587  pkCanScanPointer->waitForReconnectionThreadTrigger();
588 
589  MLOGPK(TRC, pkCanScanPointer) << "initialized reconnection thread tid= " << _tid << ", entering loop";
590  while ( true ) {
591 
592  // wait for sync: need a condition sync to step that thread once: a "trigger".
593  MLOGPK(TRC, pkCanScanPointer) << "waiting reconnection thread tid= " << _tid;
594  pkCanScanPointer->waitForReconnectionThreadTrigger();
595  MLOGPK(TRC, pkCanScanPointer)
596  << " reconnection thread tid= " << _tid
597  << " condition "<< reconnectConditionString(pkCanScanPointer->getReconnectCondition() )
598  << " action " << reconnectActionString(pkCanScanPointer->getReconnectAction())
599  << " is checked, m_failedSendCountdown= "
600  << pkCanScanPointer->getFailedSendCountdown();
601 
602  // condition
603  switch ( rcondition ){
604  case CanModule::ReconnectAutoCondition::timeoutOnReception: {
605  // no need to check the reception timeout, the control thread has done that already
606  pkCanScanPointer->resetSendFailedCountdown(); // do the action
607  break;
608  }
609  case CanModule::ReconnectAutoCondition::sendFail: {
610  pkCanScanPointer->resetTimeoutOnReception();
611  if (pkCanScanPointer->getFailedSendCountdown() > 0) {
612  continue;// do nothing
613  } else {
614  pkCanScanPointer->resetSendFailedCountdown(); // do the action
615  }
616  break;
617  }
618  // do nothing but keep counter and timeout resetted
619  case CanModule::ReconnectAutoCondition::never:
620  default:{
621  pkCanScanPointer->resetSendFailedCountdown();
622  pkCanScanPointer->resetTimeoutOnReception();
623  continue;// do nothing
624  break;
625  }
626  } // switch
627 
628  // action
629  switch ( raction ){
630  case CanModule::ReconnectAction::singleBus: {
631  MLOGPK(INF, pkCanScanPointer) << " reconnect condition " << reconnectConditionString( rcondition )
632  << " triggered action " << reconnectActionString( raction );
633 
634

636  CanModule::ms_sleep( 10000 ); // USB module boot is slow, the OS is also slow to recognize the USB back
637  break;
638  }
639

641  MLOGPK(INF, pkCanScanPointer) << " reconnect condition " << reconnectConditionString( rcondition )
642  << " triggered action " << reconnectActionString( raction );
643  break;
644  }
645  default: {
646  // we have a runtime bug
647  MLOGPK(ERR, pkCanScanPointer) << "reconnection action "
648  << (int) raction << reconnectActionString( raction )
649  << " unknown. Check your config & see documentation. No action.";

651  }
652  } // switch
653  } // while
654  MLOGPK(TRC, pkCanScanPointer) << "exiting thread...(in 2 secs)";
