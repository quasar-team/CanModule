360  inline vector<string> parseNameAndParameters(string name, string parameters){
361  LOG(Log::TRC, m_lh) << __FUNCTION__ << " name= " << name << " parameters= " << parameters;
362 
363  bool isSocketCanLinux = false;
364  std::size_t s = name.find("sock");
365  if ( s != std::string::npos ){
366  isSocketCanLinux = true;
367  }
368 
385  std::size_t foundVcan = name.find(":vcan", 0);
386  std::size_t foundSeperator = name.find_first_of (":", 0);
387  std::size_t foundPortNumber = name.find_first_of ( "0123456789", foundSeperator );
388  if ( foundPortNumber != std::string::npos ) {
389  name.erase( foundSeperator + 1, foundPortNumber - foundSeperator - 1 );
390  } else {
391  std::string rerr = "could not decode port number (need an integer somewhere, i.e. can0 or 0) in name= " + name;
392  throw std::runtime_error( rerr );
393  }
394 
395  // for socketcan, have to prefix "can" or "vcan" to port number
396  if ( isSocketCanLinux ){
397  foundSeperator = name.find_first_of (":", 0);
398  if ( foundVcan != std::string::npos ) {
399  m_sBusName = name.insert( foundSeperator + 1, "vcan");
400  } else {
401  m_sBusName = name.insert( foundSeperator + 1, "can");
402  }
403  } else {
404  m_sBusName = name;
405  }
406 
407  LOG(Log::TRC, m_lh) << __FUNCTION__ << " m_sBusName= " << m_sBusName;
408 
409  vector<string> stringVector;
410  istringstream nameSS(name);
411  string temporalString;
412  while (getline(nameSS, temporalString, ':')) {
413  stringVector.push_back(temporalString);
414  LOG(Log::TRC, m_lh) << __FUNCTION__ << " stringVector new element= " << temporalString;
415  }
416  m_CanParameters.scanParameters(parameters);
417  LOG(Log::TRC, m_lh) << __FUNCTION__ << " stringVector size= " << stringVector.size();

420  }
